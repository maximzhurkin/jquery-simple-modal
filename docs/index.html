jQuery Simple Modal
Very simple responsive jQuery modal plugin
Download
Github
Usage

Include jquery.simple-modal.css to head

<link href="/dist/css/jquery.simple-modal.min.css" rel="stylesheet">

Include jQuery and simple modal

<script src="http://code.jquery.com/jquery-latest.min.js"></script>
<script src="/dist/js/jquery.simple-modal.min.js"></script>

Initialize plugin

$(document).ready(function(){
    $('.button').click(function(){
        $().simpleModal();
    });
});

Custom settings

$().simpleModal({
    name: 'example',
    title: 'Example',
    content: '<button class="my-button">Lorem ipsum</button>',
    size: 'small',
    freeze: true,
    callback: function() {
      $('.simple-modal--example .my-button').click(function() {
        console.log('Click from modal');
      });
    }
});
